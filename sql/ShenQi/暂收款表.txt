CREATE TABLE [dbo].[T_RENT_CLAIM] ( --认领表
[ID] int IDENTITY(1, 1) NOT NULL,
[INCOME_ID] int NOT NULL,--来款ID
[AMOUNT] money NULL,--认领金额
[REASON] nvarchar(500) COLLATE Chinese_PRC_CI_AS NULL,
[IS_OPEN_INVOICE] int NULL,--0不开发票,1开发票
[STATE] int NULL, --0申请中,1财务通过,2财务驳回
[CREATE_TIME] datetime NULL,--创建时间
[CREATE_BY] int NULL,
[MODIFY_TIME] datetime NULL,--修改时间
[MODIFY_BY] int NULL,
[AUTH_TIME] datetime NULL,--审批时间
[AUTH_BY] int NULL,
[STATUS] int NULL, --0有效,-1删除的
PRIMARY KEY CLUSTERED ([ID] )
WITH ( PAD_INDEX = OFF,
FILLFACTOR = 100,
IGNORE_DUP_KEY = OFF,
STATISTICS_NORECOMPUTE = OFF,
ALLOW_ROW_LOCKS = ON,
ALLOW_PAGE_LOCKS = ON,
DATA_COMPRESSION = NONE )
 ON [PRIMARY]
)
ON [PRIMARY];
GO

CREATE TABLE [dbo].[T_RENT_REFUND] (--退款表
[ID] int IDENTITY(1, 1) NOT NULL,
[SERIAL_NUMBER] nvarchar(50) COLLATE Chinese_PRC_CI_AS NULL,--流水号
[INCOME_ID] int NULL,
[AMOUNT] money NULL,--实际退款金额
[PAYEE] nvarchar(50) COLLATE Chinese_PRC_CI_AS NULL,--收款人
[REFUND_MONEY] money NULL,--退款金额
[FEE] money NULL,--退款手续费
[REFUND_DATE] date NULL,--退款日期
[REFUND_TYPE] nvarchar(50) NULL,--退款类型  当类型是误汇款与保险退款时候  才会出险退款手续费
[PAY_WAY] nvarchar(50) COLLATE Chinese_PRC_CI_AS NULL,--支付方式
[BANK_NAME] nvarchar(50) COLLATE Chinese_PRC_CI_AS NULL,
[BANK_ACCOUNT] nvarchar(50) COLLATE Chinese_PRC_CI_AS NULL,
[REMARK] nvarchar(500) COLLATE Chinese_PRC_CI_AS NULL,
[STATE] int NULL,--0申请中,1财务通过,2财务驳回
[CREATE_TIME] datetime NULL,
[CREATE_BY] int NULL,
[MODIFY_TIME] datetime NULL,
[MODIFY_BY] int NULL,
[AUTH_TIME] datetime NULL,
[AUTH_BY] int NULL,
[STATUS] int NULL,
PRIMARY KEY CLUSTERED ([ID] )
WITH ( PAD_INDEX = OFF,
FILLFACTOR = 100,
IGNORE_DUP_KEY = OFF,
STATISTICS_NORECOMPUTE = OFF,
ALLOW_ROW_LOCKS = ON,
ALLOW_PAGE_LOCKS = ON,
DATA_COMPRESSION = NONE )
 ON [PRIMARY]
)
ON [PRIMARY];
GO
